Codeup 3180 간단한 문제

by qkrwhdah03

2020. 05. 03
in Solving

문제 출처

< 문제 >

N개의 정수로 이루어진 수열 A1,A2,…,AN 이 주어진다. 아래 두가지 간단한 질의 M를 수행해야 한다.

  • 1 i j x : Al,Al+1,…,Ar 중 x와 같은 원소의 개수를 출력한다.
  • 2 i x : Ai 를 x로 바꾼다.
입력

첫째 줄에 수열의 길이 N이 주어진다. (1 ≤ N ≤ 3∗10^5)
둘째 줄에 A1,A2,…,AN이 주어진다. (−10^9 ≤ Ai ≤ 10^9)
셋째 줄에 질의의 개수 M이 주어진다, (1 ≤ M ≤ 3∗10^5)
넷째 줄부터 쿼리가 한줄에 하나씩 주어진다. (1 ≤ i ≤ j ≤ N,−10^9 ≤ x ≤10^9)

출력

각각의 1번 질의마다 정답을 한줄에 하나씩 출력한다.

< 풀이 1 >

정말 아이디어가 떠오르지 않아서, 1번 Query는 O(log n), 2번 Query는 O(n)에 처리하는 형식의 반점짜리 풀이를 사용했다. 근데, 테스트 케이스가 약했는지 800ms 정도가 되어 정풀이 떠버리는 참사가 발생했다.
어쨌든, 일단 풀이를 소개하겠다. 일단, map<int, vector> 를 사용하여 특정 수 k가 있는 index x를 vector map[k]에 push해준다. 그러면 각 map의 vector에는 특정 수가 들어있는 좌표들이 오름차순으로 정렬되어 있는 상태이다. 그러면 1번 Query는 upper_bound와 lower_bound를 이용해서 해당 시작범위 이상, 끝 범위 초과과인 것들의 개수를 둘의 차이를 통해 구할 수 있다. 하지만, 2번 Query는 vector에 중간 삽입, 삭제를 해야하므로 O(n)의 시간복잡도가 되어버린다.

< 코드 1 >

  #include <iostream>
  #include <map>
  #include <vector>
  #include <algorithm>

  using namespace std;
  int n,m,t,a,b,c;
  map <int ,vector<int> > tree;
  vector<int> arr;
  int main()
  {
  	cin.tie(NULL);
  	ios_base::sync_with_stdio(false);

  	cin>>n;
  	arr.resize(n+5);
  	for(int i=1; i<=n; i++){
  		int tmp;
  		cin>>tmp;
  		tree[tmp].push_back(i);
  		arr[i]=tmp;
  	}
  	cin>>m;
  	for(int i=0; i<m; i++){
  		cin>>t;
  		if(t==1){
  			cin>>a>>b>>c;
  			int l=lower_bound(tree[c].begin(),tree[c].end(),a)-tree[c].begin(); //O(logn)
  			int r=upper_bound(tree[c].begin(),tree[c].end(),b)-tree[c].begin(); // O(logn)
  			cout<<r-l<<'\n';
  		}
  		else if(t==2){
  			cin>>a>>b;
  			int tmp=arr[a];
  			arr[a]=b;
  			tree[tmp].erase(lower_bound(tree[tmp].begin(),tree[tmp].end(),a)); // O(n)
  			tree[b].insert(upper_bound(tree[b].begin(),tree[b].end(),a),a); //O(n)
  		}
  	}
  }

< 풀이 2 >

이 문제를 풀기위해서는 앞선 풀이에서 설명했듯이 O(log n)에 탐색이 가능하고, O(log n)에 중간 삽입, 삭제가 가능한 형태의 알고리즘 혹은 자료구조를 사용해야한다. 정말 놀라운 것은 이런 어마어마한 자료구조가 존재했다.
바로, PBDS(Policy Based Data Structure)이다.
여기서 처음 알게되었다.
이 자료구조를 이용하면 간단히 upper_bound만 구현하면 문제가 풀려버린다.

< 코드 2 >

  #include <iostream>
  #include <map>
  #include <vector>
  #include <algorithm>
  #include <ext/pb_ds/assoc_container.hpp>
  #include <ext/pb_ds/tree_policy.hpp>
  using namespace std;
  using namespace __gnu_pbds;

  typedef tree<
  int,
  null_type,
  less<int>,
  rb_tree_tag,
  tree_order_statistics_node_update>
  ordered_set;

  int n,m,t,a,b,c;
  int arr[300001];
  map< int, ordered_set > h;

  int upper_bound(int k){
  	int r = h[c].size();
  	int l=0;
  	while(l<r){
  		int mid=(l+r)/2;
  		if(*h[c].find_by_order(mid) > k) r=mid;
  		else l=mid+1;
  	}
  	return r;
  }

  int main()
  {
  	cin.tie(NULL);
  	cout.tie(NULL);
  	ios_base::sync_with_stdio(false);

  	cin>>n;
  	for(int i=1; i<=n; i++){
  		int tmp;
  		cin>>tmp;
  		h[tmp].insert(i);
  		arr[i]=tmp;
  	}
  	cin>>m;
  	for(int i=0; i<m; i++){
  		cin>>t;
  		if(t==1){
  			cin>>a>>b>>c;
  			int l=h[c].order_of_key(a); // lower_bound
  			int r=upper_bound(b);
  			cout<<r-l<<'\n';
  		}
  		else if(t==2){
  			cin>>a>>b;
  			int tmp=arr[a];
  			arr[a]=b;
  			h[tmp].erase(a);
  			h[b].insert(a);
  		}
  	}
  }
